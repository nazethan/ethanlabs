### **Aim**
> To write SQL queries for data updating

### **Basic Syntax:**
```sql
UPDATE table_name
SET column1 = value1, column2 = value2, ...  
WHERE condition;
```

### **Example Script**
```sql title='Data Updating Operations'
-- Create a sample table for demonstration
CREATE TABLE Employees (
    EmpID INT PRIMARY KEY,
    Name VARCHAR(50),
    Department VARCHAR(30),
    Salary DECIMAL(10, 2),
    JoinDate DATE,
    City VARCHAR(25)
);

-- Insert sample data
INSERT INTO Employees VALUES 
(1, 'John Smith', 'IT', 50000.00, '2022-01-15', 'New York'),
(2, 'Alice Johnson', 'HR', 45000.00, '2021-03-20', 'Chicago'),
(3, 'Bob Wilson', 'Finance', 55000.00, '2020-07-10', 'Boston'),
(4, 'Carol Davis', 'IT', 48000.00, '2023-02-05', 'New York'),
(5, 'David Brown', 'Marketing', 42000.00, '2022-11-12', 'Los Angeles');

-- Display original data
SELECT * FROM Employees;

-- 1. Update single column (salary increase for specific employee)
UPDATE Employees 
SET Salary = 52000.00 
WHERE EmpID = 1;

-- 2. Update multiple columns simultaneously
UPDATE Employees 
SET Department = 'Senior IT', Salary = 60000.00 
WHERE EmpID = 4;

-- 3. Update based on condition (salary increase for IT department)
UPDATE Employees 
SET Salary = Salary * 1.10 
WHERE Department LIKE '%IT%';

-- 4. Update using CASE statement (conditional update)
UPDATE Employees 
SET Salary = CASE 
    WHEN Department = 'HR' THEN Salary + 5000
    WHEN Department = 'Finance' THEN Salary + 7000
    WHEN Department = 'Marketing' THEN Salary + 3000
    ELSE Salary
END;

-- 5. Update based on date condition
UPDATE Employees 
SET City = 'Remote' 
WHERE JoinDate >= '2023-01-01';

-- 6. Update using mathematical operations
UPDATE Employees 
SET Salary = ROUND(Salary * 1.05, 2) 
WHERE City = 'New York';

-- Display updated data
SELECT * FROM Employees;
```
